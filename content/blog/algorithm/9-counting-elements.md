/**
 * @param {number[]} arr
 * @return {number}
 */
var countElements = function(arr) {
    const map = new Map();
    for (let num of arr) {
        map.set(num, true);
    }
    let cnt = 0;
    for (let num of arr) {
        if (map.get(num+1)) {
           cnt++;
        }
    }
    return cnt;
};
